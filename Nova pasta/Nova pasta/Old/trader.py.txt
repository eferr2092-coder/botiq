import time
import config
from connection import get_iq

def executar_trade(par, direcao, timeframe):

    iq = get_iq()

    valor = config.VALOR_ENTRADA

    status, order_id = iq.buy(valor, par, direcao.lower(), 1)

    if not status:
        return "ERRO", 0

    print("Ordem enviada:", order_id)

    time.sleep(65)

    lucro = iq.check_win_v4(order_id)

    if lucro > 0:
        return "WIN", lucro
    else:
        return "LOSS", lucro
